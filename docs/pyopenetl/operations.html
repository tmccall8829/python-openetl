102    def write_from_dataframe(
103        self,
104        table: str,
105        df: pd.DataFrame,
106        chunksize: int = 100_000_000,
107    ) -> str:
108        """
109        Writes a pandas dataframe to a given SQL table. Note that this assumes that the
110        table we're going to write to already exists -- it will NOT create the table for
111        you.
112
113        args:
114            table: the name of the pre-existing table to write out to
115            df: the data to write
116            chunksize: the number of rows to write out at once. default to large enough
117                to write the whole df at once.
118        """
119
120        # we need to define this here instead of as a class method,
121        # bc pandas expects 4 args but as a class method it'll have 5
122        def buffer_write(table, conn, keys, data_iter):
123            """
124            Execute SQL statement inserting data (borrowed from pandas)
125
126            args:
127                table: pandas.io.sql.SQLTable
128                conn: sqlalchemy.engine.Engine or sqlalchemy.engine.Connection
129                keys: list of str column names
130                data_iter: Iterable that iterates the values to be inserted
131            """
132            # gets a DBAPI connection that can provide a cursor
133            raw_conn = conn.connection
134            with raw_conn.cursor() as cur:
135                s_buf = io.StringIO()
136                writer = csv.writer(s_buf)
137                writer.writerows(data_iter)
138                s_buf.seek(0)
139
140                if table.schema:
141                    table_name = "{}.{}".format(table.schema, table.name)
142                else:
143                    table_name = table.name
144
145                sql = f"COPY {table_name} FROM STDIN WITH CSV"
146                cur.copy_expert(sql=sql, file=s_buf)
147
148        with self.dest_conn.connect() as sql_conn:
149            with sql_conn.begin():
150                df = self.convert_column_types(df)
151                n_rows = int(df.shape[0])
152                try:
153                    df.to_sql(
154                        name=table,
155                        con=sql_conn,
156                        if_exists="append",
157                        index=False,
158                        chunksize=chunksize,
159                        method=buffer_write,
160                    )
161                except Exception as err:
162                    raise err from err
163
164                del df
165                gc.collect()
166
167        return f"Wrote {n_rows} rows to Cloud SQL table {table}"

188    def get_postgres_dtypes_for_temp_table(self, table: str) -> dict:
189        """
190        Turns a SQL table's schema into a dictionary of column names to SQLAlchemy types.
191
192        args:
193            table: name of the table schema to read
194        """
195
196        def _string_types_to_sqlachemy_types(str_type):
197            type_mapping = {
198                "ARRAY": sqlalchemy.dialects.postgresql.ARRAY,
199                "bigint": sqlalchemy.dialects.postgresql.BIGINT,
200                "boolean": sqlalchemy.dialects.postgresql.BOOLEAN,
201                "character varying": sqlalchemy.dialects.postgresql.VARCHAR,
202                "date": sqlalchemy.dialects.postgresql.DATE,
203                "double precision": sqlalchemy.dialects.postgresql.DOUBLE_PRECISION,
204                "inet": sqlalchemy.dialects.postgresql.INET,
205                "integer": sqlalchemy.dialects.postgresql.INTEGER,
206                "jsonb": sqlalchemy.dialects.postgresql.JSONB,
207                "bytea": sqlalchemy.dialects.postgresql.BYTEA,
208                "numeric": sqlalchemy.dialects.postgresql.NUMERIC,
209                "smallint": sqlalchemy.dialects.postgresql.SMALLINT,
210                "timestamp with time zone": sqlalchemy.dialects.postgresql.TSTZRANGE,
211                "text": sqlalchemy.dialects.postgresql.TEXT,
212                "timestamp without time zone": sqlalchemy.dialects.postgresql.TIMESTAMP,
213            }
214            try:
215                type_to_return = type_mapping[str_type]
216            except:
217                raise ValueError(
218                    f"Unable to map type: {str_type}. Does it exist in type_mappings?"
219                )
220
221            return type_to_return
222
223        with self.dest_conn.connect() as conn:
224            res = conn.execute(
225                f"""
226                SELECT
227                    column_name,
228                    data_type
229                FROM
230                    information_schema.columns
231                WHERE
232                    table_name = '{table}';
233                """
234            )
235            cols_to_types = res.all()
236            dtypes = {
237                i[0]: _string_types_to_sqlachemy_types(i[1]) for i in cols_to_types
238            }

315    def seed_table(self, read_table: str, read_chunksize: int, write_table: str) -> str:
316        """
317        Seeds a direct projection of a table from one DB source to another. Note that
318        this DOES create the table before writing, unlike the basic write method.
319
320        args:
321            read_table: the name of the table to read from
322            read_chunksize: the number of rows to read from read_table at a time
323            write_table: the name of the table to write out to cloud SQL. This parameter
324                allows seed_table() to seed one table (e.g., users) into Cloud SQL under
325                a different name (e.g., users_projection).
326        """
327        # delete the table we want to write out in Cloud SQL
328        self.delete_table(write_table)
329
330        if isinstance(self.source_conn, HerokuConnection):
331            reader = HerokuReader(self.source_conn)
332        elif isinstance(self.source_conn, CloudSQLConnection):
333            reader = CloudSQLReader(self.source_conn)
334        else:
335            raise TypeError(f"Unsupported source connection type {self.source_conn} for table seeding.")
336
337        write_time = datetime.datetime.now(datetime.timezone.utc)
338
339        # read from postgres
340        its = 1
341        for df in reader.table_to_dataframe(table=read_table, chunksize=read_chunksize):
342            print(f"Batch: {(its-1)*read_chunksize} to {(its)*read_chunksize}")
343            if its == 1:
344                # write to (an optionally different) table name in Cloud SQL
345                print(f"--> Creating new table {write_table} in Cloud SQL")
346                self.create_table_from_dataframe(write_table, df)
347
348            try:
349                self.write_from_dataframe(
350                    table=write_table, df=df, chunksize=read_chunksize
351                )
352            except Exception as err:
353                return f"Failed to write chunk of dataframe to {write_table}: {err}"
354
355            # explicitly mark df for garbage collection
356            del df
357            gc.collect()
358
359            its += 1
360
361        return f"Seeding of Cloud SQL table {write_table} complete in {datetime.datetime.now(datetime.timezone.utc) - write_time}"

392    def update_table_via_upsert(
393        self,
394        read_table: str,
395        data_interval_hours: int,
396        write_table: str,
397        write_table_primary_key: str,
398    ) -> str:
399        """
400        Performs an upsert of a small timeframe of data into a larger table of data with the
401        same column names and types. Note that this will not work if the table to upsert
402        has column names or a column order that does not exactly match that of the table
403        it is being upserted into.
404
405        args:
406            read_table: the name of the table to read in and use for the upsert
407            data_interval_hours: the number of hours to look into the past and read (e.g., the last 2 hours)
408            write_table: the name of the table we'll be upserting the data into
409            write_table_primary_key: the primary key of the write_table
410        """
411        with self.source_conn.connect() as read_conn:
412            write_time = datetime.datetime.now()
413
414            # read from postgres and write the temp table
415            delta_query = f"SELECT * FROM {read_table} WHERE updated_at >= (NOW() - INTERVAL'{data_interval_hours} hours')"
416            df = pd.read_sql(delta_query, read_conn)
417
418        nrows = df.shape[0]
419
420        self.append_rows_to_table_from_dataframe(
421            write_table, write_table_primary_key, df
422        )
423
424        # now, as part of the upsert process, remove any rows from our projection that have been removed from the source
425        # this is NOT dependent upon there being any new data to upsert
426        logging.info(
427            f"Upsert process: Initiating row cleanup process for table {write_table}"
428        )
429
430        def _get_projection_ids(conn, table):
431            res = conn.execute(f"SELECT id FROM {table}")
432            employment_ids = set([i[0] for i in res.all()])
433
434            return employment_ids
435
436        def _get_projection_ids_to_remove():
437            with self.dest_conn.connect() as dest:
438                with self.source_conn.connect() as source:
439                    ids = (
440                        _get_projection_ids(dest, read_table),
441                        _get_projection_ids(source, write_table),
442                    )
443                    logging.info(
444                        f"Upsert process: calculating IDs to remove from {write_table}"
445                    )
446                    deleted_ids = ids[0].difference(ids[1])
447
448            return deleted_ids
449
450        ids_to_remove = _get_projection_ids_to_remove()
451        if len(ids_to_remove) == 0:
452            logging.info(
453                f"Upsert process: no rows to remove from {read_table} projection. Exiting."
454            )
455        else:
456            logging.info(
457                f"Upsert process: removing {len(ids_to_remove)} source-deleted rows from {read_table} projection"
458            )
459            with self.dest_conn.connect() as dest:
460                # make SUPER sure we don't delete anything from Heroku PG
461                if isinstance(self.dest_conn, HerokuConnection):
462                    print(f"Cannot delete rows from Heroku PG. Exiting.")
463                else:
464                    dest.execute(
465                        f"DELETE FROM {write_table} WHERE id IN {tuple(ids_to_remove)}".replace(
466                            ",)", ")"
467                        )
468                    )
469                    logging.info(
470                        f"Upsert process: row clean-up complete for table {write_table}"
471                    )
472
473        return json.dumps(
474            {
475                "table_name": write_table,
476                "rows_added": nrows,
477                "rows_deleted": len(ids_to_remove),
478                "execution_duration": str(datetime.datetime.now() - write_time),
479                "execution_date": str(write_time),
480            }
481        )

483    def gen_update_set_parms(
484        self, merging_in: str, table_cols: list, primary_key_id: str
485    ) -> str:
486        """
487        Used to generate the parameters for an ON CONFLICT ... UPDATE SET id = S.id, ..., call.
488
489        args:
490            merging_in: the name of the table we're merging in
491            table_cols: a list of the columns in that table (which must match the
492                columns of the table we're merging into)
493            primary_key_id: the primary key of the tables we're working with (which
494                must be present in both tables)
495        """
496        update_set_parms = ", ".join(
497            [
498                f'"{col}" = {merging_in}.{col}'
499                for col in table_cols
500                if col != primary_key_id
501            ]
502        )
503
504        return update_set_parms

560    def append_rows_to_table_from_dataframe(
561        self, write_table: str, write_table_primary_key: str, df: pd.DataFrame
562    ) -> str:
563        """
564        Appends a set of rows from a dataframe to an existing table with the same schema.
565        Note that this will not work if the dataframe has column names or a column order
566        that does not exactly match that of the table it is being appended to.
567
568        args:
569        df:  dataframe containing the rows to append
570        write_table: the name of the table we'll be upserting the data into
571        write_table_primary_key: the primary key of the write_table
572        """
573
574        nrows = int(df.shape[0])
575
576        if nrows != 0:
577            temp_write_table = f"{write_table}_temp"
578            dtypes = self.get_postgres_dtypes_for_temp_table(write_table)
579
580            self.create_table_from_dataframe(temp_write_table, df, dtypes=dtypes)
581            self.write_from_dataframe(temp_write_table, df)
582
583            try:
584                with self.dest_conn.connect() as write_conn:
585                    table_cols = list(df.columns)
586                    write_conn.execute(
587                        f"""
588                            INSERT INTO {write_table}
589                            SELECT * FROM {temp_write_table}
590                            ON CONFLICT ({write_table_primary_key}) DO
591                                UPDATE SET {self.gen_update_set_parms("EXCLUDED", table_cols, write_table_primary_key)}
592                            """
593                    )
594            except Exception as err:
595                raise RuntimeError(f"Error upserting temp table: {err}") from err
596            finally:
597                self.delete_table(temp_write_table)
598
599        return f"Appending to Cloud SQL table {write_table} complete in for {nrows}."

620    def safe_insert(
621        self, table: str, schema: str = "", data: Union[list[dict], dict] = {}
622    ):
623        """
624        Inserts data (one row or many) into a table and fails if the data
625        being inserted conflicts with existing data. This depends on the
626        table being inserted into having a primary key with a unique
627        constraint.
628
629        args:
630            table: the name of the table to insert data into
631            schema (optional): the schema (e.g., postgres) for the table
632            data: the data to insert
633        """
634        start = datetime.datetime.now()
635        if isinstance(data, list):
636            cols = list(data[0].keys())
637        elif isinstance(data, dict):
638            cols = list(data.keys())
639        else:
640            raise TypeError("Error: data must be a dict or a list of dicts")
641
642        with self.dest_conn.connect() as conn:
643            try:
644                if schema == "":
645                    tbl = sqlalchemy.table(
646                        table, *[sqlalchemy.column(col) for col in cols]
647                    )
648
649                else:
650                    tbl = sqlalchemy.table(
651                        table, *[sqlalchemy.column(col) for col in cols], schema=schema
652                    )
653
654                conn.execute(sqlalchemy.insert(tbl), data)
655            except sqlalchemy.exc.IntegrityError as conflict_err:
656                raise RuntimeError(
657                    f"Error: conflict on attempted insert: {conflict_err}."
658                )
659
660        return f"Safely inserted rows in {datetime.datetime.now() - start}"

426        # this is NOT dependent upon there being any new data to upsert
